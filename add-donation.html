// add-donation.js const form = document.getElementById("donationForm"); const msg = document.getElementById("msg"); const loginWarning = document.getElementById("login-warning"); const btnSubmit = document.getElementById("btn-submit"); const elCategory = document.getElementById("category"); const elItemName = document.getElementById("item_name"); const elQuantity = document.getElementById("quantity"); const elArea = document.getElementById("area"); const elPickup = document.getElementById("pickup_location"); const elDesc = document.getElementById("description"); const imageFile = document.getElementById("image_file"); const uploadMsg = document.getElementById("upload_msg"); const imgPreview = document.getElementById("img_preview"); const btnAuto = document.getElementById("btnAutoClassify"); const aiBadge = document.getElementById("aiBadge"); const aiCategory = document.getElementById("aiCategory"); const aiConfidence = document.getElementById("aiConfidence"); const aiReason = document.getElementById("aiReason"); let uploadedImageUrl = null; let aiResult = null; function getLoggedInUser() { try { return JSON.parse(localStorage.getItem("user") || "null"); } catch { return null; } } function setMessage(text, ok = false) { msg.textContent = text || ""; msg.style.color = ok ? "#7CFCB2" : "#ffb4a2"; } function setUploadMessage(text, ok = false) { uploadMsg.textContent = text || ""; uploadMsg.style.color = ok ? "#7CFCB2" : "#ffb4a2"; } function showAI(data) { aiResult = data; aiBadge.style.display = "inline-block"; aiCategory.textContent = data.category; aiConfidence.textContent = data.confidence; aiReason.textContent = data.reason || ""; // auto fill category if option exists const opt = Array.from(elCategory.options).some((o) => o.value === data.category); if (opt) elCategory.value = data.category; } // --------------- // Auth check // --------------- const user = getLoggedInUser(); if (!user?.user_id) { loginWarning.style.display = "block"; btnSubmit.disabled = true; form.querySelectorAll("input, select, textarea, button").forEach((el) => { if (el.id !== "btnAutoClassify") el.disabled = true; }); } else { loginWarning.style.display = "none"; } // --------------- // Image upload on change // --------------- imageFile?.addEventListener("change", async () => { const file = imageFile.files?.[0]; if (!file) return; setUploadMessage("Uploading image...", true); try { const fd = new FormData(); fd.append("image", file); // server.js expects key = "image" const res = await fetch("/api/upload", { method: "POST", body: fd, }); const data = await res.json(); if (!res.ok) throw new Error(data?.message || "Upload failed"); uploadedImageUrl = data.image_url; setUploadMessage("✅ Upload success", true); imgPreview.src = uploadedImageUrl; imgPreview.style.display = "block"; } catch (e) { console.error(e); uploadedImageUrl = null; imgPreview.style.display = "none"; setUploadMessage("❌ Upload failed: " + (e.message || "unknown error")); } }); // --------------- // AI Auto Classify // --------------- btnAuto?.addEventListener("click", async () => { const item_name = elItemName.value.trim(); const description = elDesc.value.trim(); if (!item_name && !description) { aiReason.textContent = "Please enter item name or description first."; return; } aiReason.textContent = "Classifying..."; aiBadge.style.display = "none"; try { const res = await fetch("/api/donations/classify", { method: "POST", headers: { "Content-Type": "application/json" }, body: JSON.stringify({ item_name, description }), }); const data = await res.json(); if (!res.ok) throw new Error(data?.message || "Classify failed"); showAI(data); } catch (e) { console.error(e); aiReason.textContent = "❌ Classify failed: " + (e.message || "unknown error"); } }); // --------------- // Submit donation // --------------- form?.addEventListener("submit", async (e) => { e.preventDefault(); setMessage(""); const user = getLoggedInUser(); if (!user?.user_id) { setMessage("❌ You are not logged in."); return; } const payload = { donor_id: Number(user.user_id), item_name: elItemName.value.trim(), quantity: Number(elQuantity.value || 1), area: elArea.value.trim() || null, pickup_location: elPickup.value.trim() || null, description: elDesc.value.trim() || null, image_url: uploadedImageUrl || null, // AI fields (optional) category: aiResult?.category || elCategory.value || null, ai_confidence: aiResult?.confidence ?? null, ai_reason: aiResult?.reason || null, }; if (!payload.item_name || !payload.quantity) { setMessage("❌ Item Name and Quantity are required."); return; } // category is required in HTML select, so ensure it has a value if (!payload.category) { setMessage("❌ Category is required (you can choose manually or use AI Auto Classify)."); return; } try { const res = await fetch("/api/donations", { method: "POST", headers: { "Content-Type": "application/json" }, body: JSON.stringify(payload), }); const data = await res.json(); if (!res.ok) throw new Error(data?.message || "Create failed"); setMessage("✅ Donation posted successfully!", true); // If backend also returns ai, reflect it if (data?.ai?.category) showAI(data.ai); // optional: reset form.reset(); uploadedImageUrl = null; imgPreview.style.display = "none"; setUploadMessage(""); // keep AI view cleared aiResult = null; aiBadge.style.display = "none"; aiReason.textContent = ""; } catch (e) { console.error(e); setMessage("❌ Submit failed: " + (e.message || "unknown error")); } });
